% Slide 2: Introducción
\begin{frame}{Introducción y Propósito}
    \begin{columns}
        \begin{column}{0.5\textwidth}
            \begin{block}{Definición de Proyecto}
                El \textbf{USS SpiderBot} es un robot caminador cuadrúpedo de \textbf{4 grados de libertad (4-DoF)} orientado al reconocimiento autónomo en zonas de catástrofe y escombros.
            \end{block}
            \begin{block}{Desafío Tecnológico}
                Sustituir ruedas por extremidades articuladas para sortear desniveles físicos de forma estable utilizando control de bucle cerrado.
            \end{block}
        \end{column}
        \begin{column}{0.5\textwidth}
            \begin{center}
                \includegraphics[width=0.95\textwidth]{spiderbot_render.png}
                \\ \vspace{0.2cm}
                \small\color{USSBlue}\textit{Renderizado 3D de la estructura mecánica}
            \end{center}
        \end{column}
    \end{columns}
\end{frame}

% Slide 3: Diseño Mecánico
\begin{frame}{Diseño Mecánico y Estructural (OpenSCAD)}
    \begin{columns}
        \begin{column}{0.5\textwidth}
            \begin{block}{Chasis Doble Deck}
                \begin{itemize}
                    \item \textbf{Placa Inferior ($r=65\text{mm}$):} Aloja perimetralmente los brackets de cadera.
                    \item \textbf{Placa Superior ($r=65\text{mm}$):} Cuenta con cuna central para protoboard de 400 puntos y marcos protectores.
                \end{itemize}
            \end{block}
            \begin{block}{Configuración Cinemática}
                Juntas en paralelo horizontal (\textbf{Pitch-Pitch}) para maximizar torque vertical y evitar cizalle dinámico.
            \end{block}
        \end{column}
        \begin{column}{0.5\textwidth}
            \begin{alertblock}{Tolerancias de Precisión (FDM)}
                \begin{itemize}
                    \item \textbf{Cavidad Servo:} $+0.3\text{mm}$ de holgura nominal.
                    \item \textbf{Bolsillos de Doble Profundidad:} Dual-depth ($30\text{mm}$ abajo para cables, $25\text{mm}$ arriba).
                    \item \textbf{Single-Arm Horn Locks:} Ranuras chavetadas que encastran el horn estriado a presión.
                \end{itemize}
            \end{alertblock}
        \end{column}
    \end{columns}
\end{frame}

% Slide 4: Arquitectura Electrónica
\begin{frame}{Arquitectura Electrónica y Conexiones}
    \begin{columns}
        \begin{column}{0.48\textwidth}
            \begin{block}{Bus I2C y Señales PWM}
                El microcontrolador \textbf{ESP32 DevKit V1} comanda directamente las señales y buses:
                \begin{enumerate}
                    \item \textbf{MPU6050 ($0x68$):} Lectura I2C de la IMU inercial de 6 ejes.
                    \item \textbf{4x Servos PWM:} Control de posición directo por caderas en GPIOs (13, 15, 4, 23).
                \end{enumerate}
            \end{block}
        \end{column}
        \begin{column}{0.52\textwidth}
        \begin{alertblock}{Alimentación Regulada Simplificada (LM2596)}
            \begin{itemize}
                \item \textbf{Pack 2x18650 (7.4V):} Un único portapilas dual provee la tensión de entrada al regulador.
                \item \textbf{LM2596 (Regulado a 5.0V):} Un único módulo reductor step-down de alta eficiencia.
                \item \textbf{Alimentación en Paralelo:} Suministro de 5.0V comunes para la ESP32 (lógica), sensores y actuadores (servos).
                \item \textbf{GND Unificado:} Masa de referencia común para control y potencia.
            \end{itemize}
        \end{alertblock}
        \end{column}
    \end{columns}
\end{frame}

% Slide 4b: IA Sensorial Embebida
\begin{frame}{IA Sensorial Embebida en ESP32}
    \begin{columns}
        \begin{column}{0.48\textwidth}
            \begin{block}{Clasificador por Árbol de Decisión}
                Ejecuta en tiempo real ($20\text{ Hz}$) sobre valores brutos del MPU6050:
                \begin{itemize}
                    \item \textbf{FALLEN:} Inclinación $> \pm 45^\circ$, caída libre ($|a| < 0.2g$) o impacto ($> 2.0g$).
                    \item \textbf{PUSHED:} Reposo con giroscopio $> 120^\circ$/s (empuje externo).
                    \item \textbf{SLIPPING:} Marcha sin variación inercial (patas en vacío).
                    \item \textbf{NORMAL:} Operación estable.
                \end{itemize}
            \end{block}
        \end{column}
        \begin{column}{0.52\textwidth}
            \begin{alertblock}{Reacción Autónoma en el Bucle}
                \begin{itemize}
                    \item Si \textbf{FALLEN}: Se aborta la marcha, los servos vuelven a pose segura y el buzzer emite alarma pulsante.
                    \item Estado transmitido al dashboard web vía API \texttt{/telemetry} con indicación visual por colores (rojo intermitente para caída, naranja para perturbación/derrape, verde para normal).
                    \item Sin falsos positivos: los umbrales se calibran para ignorar inclinación leve de la marcha de gateo.
                \end{itemize}
            \end{alertblock}
        \end{column}
    \end{columns}
\end{frame}

% Slide 5: Firmware Multitarea
\begin{frame}{Firmware Multitarea Asíncrono (\codigo{uasyncio})}
    \begin{columns}
        \begin{column}{0.5\textwidth}
            \begin{block}{Planificación Cooperativa}
                Sustitución de llamadas bloqueantes (\codigo{time.sleep\_ms()}) por corrutinas no bloqueantes concurrentes:
                \begin{itemize}
                    \item \codigo{sensor\_updater()}: Adquisición MPU6050 y Sonar a $20\text{ Hz}$.
                    \item \codigo{locomotion\_loop()}: Lazo de cinemática y control servo.
                    \item \codigo{start\_server\_task()}: Socket TCP HTTP del dashboard.
                \end{itemize}
            \end{block}
        \end{column}
        \begin{column}{0.5\textwidth}
            \begin{alertblock}{Estado Compartido y Seguridad}
                \begin{itemize}
                    \item \textbf{state.py:} Singleton global que comparte la telemetría y órdenes de control en tiempo real.
                    \item \textbf{Lazo de Seguridad:} Si el sonar detecta un obstáculo a menos de $15\text{ cm}$, se cancela la locomoción en menos de $25\text{ ms}$, regresando a pose segura con pitido acústico.
                \end{itemize}
            \end{alertblock}
        \end{column}
    \end{columns}
\end{frame}

% Slide 6: Marcha de Gateo
\begin{frame}{Algoritmo de Locomoción: Marcha de Gateo 4-DoF}
    \begin{columns}
        \begin{column}{0.5\textwidth}
            \begin{block}{Oscilación Desfasada de Caderas}
                Estabilidad estática óptima: el centro de gravedad reside de manera segura en el polígono de sustentación de 3 patas mientras la cuarta pata desliza su cadera en el aire.
            \end{block}
            \begin{block}{Ciclo Cinemático (5 Fases)}
                Secuencia repetitiva: Paso Pata FR $\rightarrow$ Paso Pata RR $\rightarrow$ Paso Pata FL $\rightarrow$ Paso Pata RL $\rightarrow$ Empuje general del cuerpo hacia adelante (retroceso de todas las caderas).
            \end{block}
        \end{column}
        \begin{column}{0.5\textwidth}
            \begin{center}
                \includegraphics[width=0.95\textwidth]{gait_simulation.png}
                \\ \vspace{0.1cm}
                \small\color{USSBlue}\textit{Perfiles de articulación de cadera asíncronos}
            \end{center}
        \end{column}
    \end{columns}
\end{frame}

% Slide 7: Optimización y Rediseño a 4-DoF
\begin{frame}{Optimización y Rediseño a 4-DoF}
    \begin{columns}
        \begin{column}{0.55\textwidth}
            \begin{block}{Justificación del Rediseño}
                Ante restricciones de tiempo de fabricación y fatiga de materiales en rodillas, se eliminaron los 4 servos fémur, fijando mecánicamente las rodillas a $90^\circ$ (escuadra rígida).
            \end{block}
            \begin{block}{Ventajas del Sistema Simplificado}
                \begin{itemize}
                    \item \textbf{Consumo Eléctrico:} Reducción del $50\%$ en la corriente pico (de $3.0\text{ A}$ a $1.5\text{ A}$).
                    \item \textbf{Carga Útil:} Mayor autonomía de batería y menor peso muerto del robot.
                    \item \textbf{Failsafe Activo:} La IMU sigue vigilando la estabilidad y gatilla parada segura por IA ante colisiones.
                \end{itemize}
            \end{block}
        \end{column}
        \begin{column}{0.45\textwidth}
            \begin{center}
                \includegraphics[width=0.95\textwidth]{stabilization_response.png}
                \\ \vspace{0.1cm}
                \small\color{USSBlue}\textit{Simulación del comportamiento transitorio inercial}
            \end{center}
        \end{column}
    \end{columns}
\end{frame}
